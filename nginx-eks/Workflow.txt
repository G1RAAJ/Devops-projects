🔷 2. EC2 (Jenkins Server Setup)
Install tools (Amazon Linux 2023)

# update
sudo dnf update -y && sudo yum install -y git wget curl unzip tar vim

# Java
sudo dnf install java-17-amazon-corretto -y

# Jenkins
sudo wget -O /etc/yum.repos.d/jenkins.repo \
https://pkg.jenkins.io/redhat-stable/jenkins.repo

sudo rpm --import https://pkg.jenkins.io/redhat-stable/jenkins.io-2023.key
sudo dnf install jenkins -y
sudo systemctl enable jenkins
sudo systemctl start jenkins
sudo systemctl status jenkins

# Docker
sudo dnf install docker -y
sudo systemctl start docker
sudo usermod -aG docker jenkins

# AWS CLI
sudo dnf install awscli -y

# kubectl
curl -o kubectl https://amazon-eks.s3.us-east-1.amazonaws.com/latest/2023-11-14/bin/linux/amd64/kubectl
chmod +x kubectl
sudo mv kubectl /usr/local/bin/

# eksctl
curl --silent --location "https://github.com/weaveworks/eksctl/releases/latest/download/eksctl_Linux_amd64.tar.gz" | tar xz
sudo mv eksctl /usr/local/bin/

# Terraform
sudo dnf install -y yum-utils
sudo yum-config-manager --add-repo https://rpm.releases.hashicorp.com/AmazonLinux/hashicorp.repo
sudo dnf install terraform -y

Problems solving Commands
-----------------------------
1.Force delete using AWS CLI
aws eks delete-cluster --name myeks --region us-east-1

2.Monitor deletion
aws eks describe-cluster --name myeks --region us-east-1

3.List node groups
aws eks list-nodegroups \
--cluster-name myeks \
--region us-east-1

4.Delete node group
aws eks delete-nodegroup \
--cluster-name myeks \
--nodegroup-name default \
--region us-east-1

5.Wait until deletion completes
aws eks describe-nodegroup \
--cluster-name myeks \
--nodegroup-name default \
--region us-east-1

6.Verify
aws eks list-clusters
[] # means emty cluster

7.Get your real values
aws ec2 describe-vpcs --filters "Name=isDefault,Values=true" \
--query "Vpcs[0].VpcId" --output text

aws ec2 describe-subnets \
--filters "Name=default-for-az,Values=true" \
--query "Subnets[*].SubnetId" --output text

8.Delete Load Balancers
aws elbv2 describe-load-balancers

9.Delete NAT Gateway
aws ec2 describe-nat-gateways

aws ec2 delete-nat-gateway --nat-gateway-id nat-xxxx

10.Detach & delete Internet Gateway
aws ec2 describe-internet-gateways
aws ec2 detach-internet-gateway --internet-gateway-id igw-xxxx --vpc-id vpc-xxxx
aws ec2 delete-internet-gateway --internet-gateway-id igw-xxxx

11.Delete Subnets
aws ec2 describe-subnets --filters Name=vpc-id,Values=vpc-xxxx
aws ec2 delete-subnet --subnet-id subnet-xxxx

12.Finally delete VPC
aws ec2 delete-vpc --vpc-id vpc-xxxx
