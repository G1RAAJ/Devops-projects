Push code to GitHub
Jenkins pulls code → builds Docker image
Push image to AWS ECR
Deploy to Kubernetes (EKS or self-managed cluster)
------------------------------------------------------
# One-Time Setup on Jenkins (Amazon Linux 2023)
sudo dnf update -y

# Core tools
sudo dnf install -y git docker unzip curl

# Start Docker
sudo systemctl start docker
sudo systemctl enable docker

# Add permissions
sudo usermod -aG docker ec2-user
sudo usermod -aG docker jenkins

# Install AWS CLI
curl "https://awscli.amazonaws.com/awscli-exe-linux-x86_64.zip" -o awscliv2.zip
unzip awscliv2.zip
sudo ./aws/install

# Install kubectl
curl -LO "https://dl.k8s.io/release/$(curl -L -s https://dl.k8s.io/release/stable.txt)/bin/linux/amd64/kubectl"
chmod +x kubectl
sudo mv kubectl /usr/local/bin/

# Verify tools
git --version
docker --version
aws --version
kubectl version --client

# Configure AWS Access look like 652942059461
aws sts get-caller-identity

# Create ECR repository
aws ecr create-repository \
--repository-name flask-app \
--region us-east-1

# Verify repository look like "repositoryName": "flask-app"
aws ecr describe-repositories --region us-east-1

# Login to ECR
ACCOUNT_ID=652942059461
REGION=us-east-1

aws ecr get-login-password --region $REGION | \
docker login --username AWS \
--password-stdin $ACCOUNT_ID.dkr.ecr.$REGION.amazonaws.com

# Push again
docker push 652942059461.dkr.ecr.us-east-1.amazonaws.com/flask-app:9

# To know all cluster names
aws eks list-clusters
aws eks list-clusters --region us-east-1

# Install eksctl
curl --silent --location \
"https://github.com/weaveworks/eksctl/releases/latest/download/eksctl_$(uname -s)_amd64.tar.gz" \
| tar xz -C /tmp

sudo mv /tmp/eksctl /usr/local/bin
eksctl version

# Verify AWS access
aws sts get-caller-identity

# Create EKS Cluster
eksctl create cluster \
--name myeks \
--region us-east-1 \
--nodegroup-name standard-workers \
--node-type t3.medium \
--nodes 2

⏳ Wait Time:  10–15 minutes

# Verify cluster
aws eks list-clusters --region us-east-1

# output should be below
{
  "clusters": ["myeks"]
}

# Connect to EKS
aws eks update-kubeconfig \
--region us-east-1 \
--name myeks

# Check real cluster status
aws eks describe-cluster \
--name myeks \
--region us-east-1 \
--query "cluster.status"

# Check nodegroup
eksctl get nodegroup \
--cluster myeks \
--region us-east-1

kubectl get nodes

# Create nodegroup
eksctl create nodegroup \
--cluster myeks \
--region us-east-1 \
--name ng-1 \
--node-type t3.small \
--nodes 2

kubectl get nodes

# if cluster not properly created and recreate it
eksctl delete cluster --region=us-east-1 --name=myeks

Jenkins → connects to cluster → deploys app
------------

# Build & Push
docker build -t flask-app .
docker tag flask-app:latest <ACCOUNT_ID>.dkr.ecr.us-east-1.amazonaws.com/flask-app:latest
docker push <ACCOUNT_ID>.dkr.ecr.us-east-1.amazonaws.com/flask-app:latest

# Deploy to Kubernetes
kubectl apply -f k8s/deployment.yaml
kubectl apply -f k8s/service.yaml

kubectl get pods
kubectl get svc

# Install Jenkins (Ubuntu EC2)
sudo apt update
sudo apt install openjdk-17-jdk -y

wget -q -O - https://pkg.jenkins.io/debian/jenkins.io.key | sudo apt-key add -
sudo sh -c 'echo deb http://pkg.jenkins.io/debian binary/ > /etc/apt/sources.list.d/jenkins.list'

sudo apt update
sudo apt install jenkins -y
sudo systemctl start jenkins
