Jenkins Built in node memory configuration
--------------------------------------------
Free Swap Space 2GB
⚙️ PART 1 — Set 2GB Swap
1️⃣ Remove existing swap (if any)
sudo swapoff -a
sudo rm -f /swapfile

2️⃣ Create new 2GB swap
sudo fallocate -l 2G /swapfile

3️⃣ Set permissions
sudo chmod 600 /swapfile

4️⃣ Create swap area
sudo mkswap /swapfile

5️⃣ Enable swap
sudo swapon /swapfile

6️⃣ Verify
free -h

7️⃣ Make it permanent
echo '/swapfile swap swap defaults 0 0' | sudo tee -a /etc/fstab

⚙️ PART 2 — Set /tmp to 2GB
1️⃣ Check current mount
mount | grep /tmp

2️⃣ Edit fstab
sudo nano /etc/fstab

👉Add this line:
tmpfs /tmp tmpfs defaults,size=2G 0 0

to save it, press Ctrl+x, y and m

3️⃣ Apply changes
sudo mount -o remount /tmp

4️⃣ Verify
df -h /tmp

set each 1GB as required on jenkins/nodes/Built-in-node/configure
Apply and Save

pipeline {
    agent any

    environment {
        AWS_REGION   = "us-east-1"
        CLUSTER_NAME = "myeks"
    }

    stages {

        stage('Checkout Repo') {
            steps {
                git branch: 'main', url: 'https://github.com/G1RAAJ/Devops-projects.git'
            }
        }

        stage('Update Kubeconfig') {
            steps {
                sh '''
                aws eks update-kubeconfig \
                --region $AWS_REGION \
                --name $CLUSTER_NAME || true
                '''
            }
        }

        stage('Delete Kubernetes Resources') {
            steps {
                sh '''
                echo "Deleting all Kubernetes resources..."
                kubectl delete all --all || true
                '''
            }
        }

        stage('Destroy EKS Cluster') {
            steps {
                sh '''
                echo "Destroying EKS cluster..."
                eksctl delete cluster \
                --name $CLUSTER_NAME \
                --region $AWS_REGION
                '''
            }
        }
    }

    post {
        success {
            echo "🔥 Cluster and resources deleted successfully"
        }
        failure {
            echo "❌ Destroy failed"
        }
        always {
            cleanWs()
        }
    }
}
