Resource deleting Commands
-----------------------------
1.Force delete using AWS CLI
aws eks delete-cluster --name myeks --region us-east-1

2.Monitor deletion
aws eks describe-cluster --name myeks --region us-east-1

3.List node groups
aws eks list-nodegroups \
--cluster-name myeks \
--region us-east-1

4.Delete node group
aws eks delete-nodegroup \
--cluster-name myeks \
--nodegroup-name default \
--region us-east-1

5.Wait until deletion completes
aws eks describe-nodegroup \
--cluster-name myeks \
--nodegroup-name default \
--region us-east-1

6.Verify
aws eks list-clusters
[] # means emty cluster

7.Get your real values
aws ec2 describe-vpcs --filters "Name=isDefault,Values=true" \
--query "Vpcs[0].VpcId" --output text

aws ec2 describe-subnets \
--filters "Name=default-for-az,Values=true" \
--query "Subnets[*].SubnetId" --output text

8.Delete Load Balancers
aws elbv2 describe-load-balancers

9.Delete NAT Gateway
aws ec2 describe-nat-gateways

aws ec2 delete-nat-gateway --nat-gateway-id nat-xxxx

10.Detach & delete Internet Gateway
aws ec2 describe-internet-gateways
aws ec2 detach-internet-gateway --internet-gateway-id igw-xxxx --vpc-id vpc-xxxx
aws ec2 delete-internet-gateway --internet-gateway-id igw-xxxx

11.Delete Subnets
aws ec2 describe-subnets --filters Name=vpc-id,Values=vpc-xxxx
aws ec2 delete-subnet --subnet-id subnet-xxxx

12.Finally delete VPC
aws ec2 delete-vpc --vpc-id vpc-xxxx

Jenkins Built in node memory configuration
--------------------------------------------
Free Swap Space 2GB
⚙️ PART 1 — Set 2GB Swap
1️⃣ Remove existing swap (if any)
sudo swapoff -a
sudo rm -f /swapfile

2️⃣ Create new 2GB swap
sudo fallocate -l 2G /swapfile

3️⃣ Set permissions
sudo chmod 600 /swapfile

4️⃣ Create swap area
sudo mkswap /swapfile

5️⃣ Enable swap
sudo swapon /swapfile

6️⃣ Verify
free -h

7️⃣ Make it permanent
echo '/swapfile swap swap defaults 0 0' | sudo tee -a /etc/fstab

⚙️ PART 2 — Set /tmp to 2GB
1️⃣ Check current mount
mount | grep /tmp

2️⃣ Edit fstab
sudo nano /etc/fstab

👉Add this line:
tmpfs /tmp tmpfs defaults,size=2G 0 0

to save it, press Ctrl+x, y and m

3️⃣ Apply changes
sudo mount -o remount /tmp

4️⃣ Verify
df -h /tmp

set each 1GB as required on jenkins/nodes/Built-in-node/configure
Apply and Save

// 🔥 Auto Cleanup AFTER 1 hour
        cleanup {
            script {
                echo "Waiting 1 hour before cleanup..."
                sleep(time: 1, unit: 'HOURS')

                withCredentials([[
                    $class: 'AmazonWebServicesCredentialsBinding',
                    credentialsId: 'aws-creds'
                ]]) {

                    sh '''
                    aws eks update-kubeconfig \
                    --region $AWS_REGION \
                    --name $CLUSTER_NAME || true

                    echo "===== DELETE K8S RESOURCES ====="
                    kubectl delete all --all || true
                    '''

                    dir('Nginx_eks/Terraform') {
                        sh '''
                        terraform init
                        terraform destroy -auto-approve
                        '''
                    }
                }
            }
        }
