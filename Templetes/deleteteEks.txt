Jenkins Built in node memory configuration
--------------------------------------------
Free Swap Space 2GB
⚙️ PART 1 — Set 2GB Swap
1️⃣ Remove existing swap (if any)
sudo swapoff -a
sudo rm -f /swapfile

2️⃣ Create new 2GB swap
sudo fallocate -l 2G /swapfile

3️⃣ Set permissions
sudo chmod 600 /swapfile

4️⃣ Create swap area
sudo mkswap /swapfile

5️⃣ Enable swap
sudo swapon /swapfile

6️⃣ Verify
free -h

7️⃣ Make it permanent
echo '/swapfile swap swap defaults 0 0' | sudo tee -a /etc/fstab

⚙️ PART 2 — Set /tmp to 2GB
1️⃣ Check current mount
mount | grep /tmp

2️⃣ Edit fstab
sudo nano /etc/fstab

👉Add this line:
tmpfs /tmp tmpfs defaults,size=2G 0 0

to save it, press Ctrl+x, y and m

3️⃣ Apply changes
sudo mount -o remount /tmp

4️⃣ Verify
df -h /tmp

set each 1GB as required on jenkins/nodes/Built-in-node/configure
Apply and Save

--name myeks-1 \
--region us-east-1 \
--query "cluster.status"

aws eks list-nodegroups \
--cluster-name myeks-1 \
--region us-east-1

Detailed status
aws eks describe-nodegroup \
--cluster-name myeks-1 \
--nodegroup-name default \
--region us-east-1 \
--query "nodegroup.status"

aws ec2 describe-instances \
--filters "Name=tag:eks:cluster-name,Values=myeks-1" \
--query "Reservations[*].Instances[*].InstanceId" \
--region us-east-1

✅ 🔍 4. Check ECR Repository
aws ecr describe-repositories --region us-east-1

✅ 🔍 5. Check Kubernetes Access

✅ 🔍 7. Check Service

To delete all resources
-----------------------------

🔧STEP 1: Verify AWS identity
aws sts get-caller-identity

🔧STEP 2: Delete ALL nodegroups
List nodegroups
aws eks list-nodegroups \
--cluster-name myeks-1 \
--region us-east-1

🔧 STEP 2: Delete ALL nodegroups
for ng in $(aws eks list-nodegroups \
--cluster-name myeks-1 \
--region us-east-1 \
--query 'nodegroups[]' \
--output text); do
  echo "Deleting $ng"
  aws eks delete-nodegroup \
  --cluster-name myeks-1 \
  --nodegroup-name $ng \
  --region us-east-1
done

-> Wait until deleted
aws eks list-nodegroups \
--cluster-name myeks-1 \
--region us-east-1

👉 Output must be:

[]

🔧 STEP 3: Delete EKS cluster
aws eks delete-cluster \
--name myeks-1 \
--region us-east-1

🔧 STEP 4: Delete ECR repository
List repos:
aws ecr describe-repositories --region us-east-1

Delete repo
aws ecr delete-repository \
--repository-name nginx \
--region us-east-1 \
--force

🔧 STEP 5: Cleanup Terraform
cd nginx-myeks/Tform

Destroy infra
terraform destroy -auto-approve

Remove local files
rm -rf .terraform*
rm -f terraform.tfstate*

🔧 STEP 6: Remove kubeconfig
rm -rf ~/.kube/config

🔧 STEP 7: Remove Docker images
docker rmi nginx -f

🔧 STEP 8: Remove AWS credentials
rm -rf ~/.aws/credentials
rm -rf ~/.aws/config

✅ Final One-Line Cleanup Script
aws eks list-nodegroups --cluster-name myeks-1 --region us-east-1 --query 'nodegroups[]' --output text | xargs -I {} aws eks delete-nodegroup --cluster-name myeks-1 --nodegroup-name {} --region us-east-1
